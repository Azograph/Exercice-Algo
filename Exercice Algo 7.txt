• Mini projet 1 : jeu du nombre mystère

mystère <- générer aléatoirement un nombre entre 1 et 100
DEMANDER PROPOSTION
proposition <- entrée par l'utilisateur
POUR i de 1 à 3
	SI proposition = mystère
		AFFICHER "VICTOIRE !" + i 
	SINON SI  proposition < mystère
		AFFICHER "Trop petit" + i 
	SINON SI proposition > mystère 
		AFFICHER "Trop grand" + i 
	SINON SI i = 3 
		AFFICHER "tentative épuisées"
	SINON SI proposition =< 0
		AFFICHER "erreur"
	SINON SI proposition =>101
		AFFICHER "erreur"
	FIN SI 
FIN POUR
